import os def gravaInformacao(): nome = input('Informe o seu nome: ') idade = input('Digite sua idade: ') telefone = input('Digite seu telefone: ') with open('pessoa.txt', 'w', encoding='utf-8') as pessoa: pessoa.write(nome + '\n') with open('data.txt', 'w', encoding='utf-8') as data: data.write(idade + '\n') with open('fone.txt', 'w', encoding='utf-8') as fone: fone.write(telefone + '\n') def gravaHabilidade(): resposta = 's' while resposta == 's': habil = input('Quais são suas habilidades? ') with open('habilidade.txt', 'a', encoding='utf-8') as habilidade_file: habilidade_file.write(habil + '\n') resposta = input('Deseja continuar adicionando habilidades (S/N)? ') def gravaTrabalho(): resposta = 's' while resposta == 's' or resposta == 'S': trab = input('Digite onde já trabalhou: ') with open('trabalho.txt', 'a', encoding='utf-8') as trabalho_file: trabalho_file.write(trab + '\n') resposta = input('Deseja continuar adicionando experiência (S/N)? ') def criacaoCurriculo(): def ler_linhas(m): with open(m, 'r', encoding='utf-8') as f: return [line.strip() for line in f.readlines() if line.strip()] nomes = ler_linhas('pessoa.txt') idades = ler_linhas('data.txt') telefones = ler_linhas('fone.txt') habilidades = ler_linhas('habilidade.txt') trabalhos = ler_linhas('trabalho.txt') habilidades_html = "".join([f'
  • {h}
    • \n' for h in habilidades]) if habilidades else '
  • Nenhuma habilidade informada.
    • ' trabalhos_html = "".join([f'
  • {t}
    • \n' for t in trabalhos]) if trabalhos else '
  • Nenhuma experiência de trabalho informada.
    • ' linhaCurriculo = f""" Currículo de {nomes}

    {nomes}

    Informações Pessoais

    Idade: {idades}

    Telefone: {telefones}

    Habilidades

      {habilidades_html}

    Experiência Profissional

      {trabalhos_html}
    """ curriculoHtml = 'curriculo.html' with open(curriculoHtml, 'w', encoding='utf-8') as curriculo: curriculo.write(linhaCurriculo) print("Você pode abrir este arquivo no seu navegador para visualizar.") if __name__ == "__main__": gravaInformacao() gravaHabilidade() gravaTrabalho() criacaoCurriculo()

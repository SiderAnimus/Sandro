import webbrowser # 1. Coleta de Informações nome = input("Nome Completo: ") idade = input("Idade: ") telefone = input("Telefone: ") url_foto = input("URL da sua foto de perfil (deixe em branco para nenhuma): ") # Nova entrada habilidades = input("Habilidades (separadas por vírgula): ").split(',') trabalhos = input("Trabalhos Anteriores (separados por ponto e vírgula): ").split(';') # 2. Geração do Conteúdo HTML # Prepara listas HTML habilidades_html = "".join([f"
  • {h.strip()}
    • " for h in habilidades if h.strip()]) # Garante que não adiciona itens vazios trabalhos_html = "".join([f"
  • {t.strip()}
    • " for t in trabalhos if t.strip()]) # Garante que não adiciona itens vazios # Condicionalmente adiciona a imagem se o URL for fornecido foto_html = f'' if url_foto else '' # Estrutura HTML completa usando f-string multilinha (tripla aspas) html_content = f""" Currículo de {nome}

    {nome}

    {foto_html}

    Idade: {idade}

    Telefone: {telefone}

    Habilidades

      {habilidades_html}

    Experiência Profissional

      {trabalhos_html}
    """ # 3. Escrita no Arquivo e Abertura nome_arquivo = "curriculo_com_foto.html" with open(nome_arquivo, "w", encoding="utf-8") as file: file.write(html_content) print(f"\nCurrículo gerado com sucesso! Arquivo: {nome_arquivo}") # Abrir o arquivo no navegador (simula o acesso por link local) webbrowser.open_new_tab(nome_arquivo)
